# 💻 דוגמה בסיסית לקוד React

```javascript
import React, { useState } from "react";

const megillah = [
  { text: "ויהי בימי אחשורוש...", character: "story" },
  { text: "ויבוא המן...", character: "haman" },
  { text: "ותלבש אסתר מלכות", character: "esther" }
];

export default function Megillah() {
  const [noise, setNoise] = useState(false);

  const handleLine = (line) => {
    if (line.character === "haman") {
      setNoise(true);
      setTimeout(() => setNoise(false), 1500);
    }
  };

  return (
    <div className="scroll">
      {megillah.map((line, i) => (
        <p key={i} onClick={() => handleLine(line)}>
          {line.text}
        </p>
      ))}

      {noise && <div className="gragger">🔊 רעשן!!!</div>}
    </div>
  );
}
```
